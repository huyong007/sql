- 递归
```js
var reverseString = function (s) {
    s = recurFun(0, s.length - 1, s.length, s);
    console.log(s,'s');
    
    return s;
};

function recurFun(left, right, len, arrS) {
    if (len === 0 || len === 1) {
        return arrS;
    }
    let leftValue = arrS[left];
    arrS[left] = arrS[right];
    arrS[right] = leftValue;
    left += 1;
    right -= 1
    len -= 2;
    if (len > 1) {
        return recurFun(left, right, len, arrS);
    } else {
        return arrS;
    }
}
reverseString(["h", "e", "l", "l", "o"]);
```

- 双指针方法
```js
var reverseString = function (s) {
    let left = 0, right = s.length - 1;
    while (left < right) {
        [s[left], s[right]] = [s[right], s[left]];
        left++;
        right--;
    }
    return s;
};
reverseString(["h", "e", "l", "l", "o"]);
```
