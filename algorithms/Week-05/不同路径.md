> 62.不同路径
```js
var uniquePaths = function (m, n) {
    let mArr = new Array(m);
    let dpArray = [];
    for (let i = 0; i < n; ++i) {
        dpArray[i] = [].concat(mArr);
        dpArray[i][m - 1] = 1;
    }
    for (let j = 0; j < m; j++) {
        dpArray[n - 1][j] = 1;
    }
    for (let k = n - 1; k >= 0; --k) {
        for (let l = m - 1; l >= 0; --l) {
            if (!dpArray[k][l]) {
                dpArray[k][l] = dpArray[k + 1][l] + dpArray[k][l + 1];
            }
        }
    }
    return dpArray[0][0];
};

uniquePaths(5, 3);
```
